<script>
    let timer;
    let time = 0;
    let temperature = 0;
    let running = false;

    const ctx = document.getElementById('tempChart').getContext('2d');
    const chart = new Chart(ctx, {
        type: 'line',
        data: {
            labels: [0],
            datasets: [{
                label: 'กราฟอุณหภูมิกับเวลา',
                data: [0],
                borderColor: '#3498db',
                backgroundColor: 'rgba(52, 152, 219, 0.1)',
                fill: true,
                tension: 0.3
            }]
        },
        options: {
            responsive: true,
            scales: {
                x: { title: { display: true, text: 'เวลา (นาที)' }, min: 0, max: 30 },
                y: { title: { display: true, text: 'อุณหภูมิ (°C)' }, min: -5, max: 110 }
            }
        }
    });

    function createIce() {
        const container = document.getElementById('iceContainer');
        container.innerHTML = '';
        for(let i=0; i<15; i++) {
            const d = document.createElement('div');
            d.className = 'ice-cube';
            container.appendChild(d);
        }
    }

    function update() {
        if (time >= 30) {
            clearInterval(timer);
            document.getElementById('flame').style.display = 'none';
            return;
        }
        
        time++;
        
        // Scientific Logic:
        // 0-6 min: Melting (0 deg)
        // 6-22 min: Heating (Up to 100 deg)
        // 22-30 min: Boiling (100 deg)
        
        if (time <= 6) {
            temperature = 0;
            document.getElementById('status').innerText = "กำลังละลาย (ของแข็ง + ของเหลว)";
            document.getElementById('iceContainer').style.opacity = 1 - (time/6);
            document.getElementById('water').style.height = (time * 10) + "%";
        } else if (temperature < 100) {
            temperature += 6.5; 
            if(temperature > 100) temperature = 100;
            document.getElementById('status').innerText = "ของเหลว (น้ำ)";
            document.getElementById('iceContainer').style.display = 'none';
            document.getElementById('water').style.height = "65%";
        } else {
            temperature = 100;
            document.getElementById('status').innerText = "กำลังเดือด (ของเหลว + แก๊ส)";
        }

        // Update UI
        document.getElementById('timeText').innerText = time;
        document.getElementById('tempText').innerText = temperature.toFixed(1);
        document.getElementById('mercury').style.height = (temperature * 0.8 + 10) + "%";
        
        // Update Chart
        chart.data.labels.push(time);
        chart.data.datasets[0].data.push(temperature);
        chart.update();
    }

    document.getElementById('btnStart').addEventListener('click', function() {
        if(!running) {
            running = true;
            this.disabled = true;
            document.getElementById('flame').style.display = 'block';
            timer = setInterval(update, 800); // 1 step per 0.8s
        }
    });

    document.getElementById('btnReset').addEventListener('click', function() {
        location.reload();
    });

    createIce();
</script>
