<script>

const canvas = document.getElementById("labCanvas");
const ctx = canvas.getContext("2d");

let time = 0;
let temp = -10;
let heatPower = 1;
let interval = null;

let chartData = [];
let labels = [];

const chart = new Chart(document.getElementById('chartCanvas'), {
type:'line',
data:{
labels:labels,
datasets:[{
label:'อุณหภูมิ',
data:chartData,
fill:false,
borderWidth:3
}]
},
options:{
responsive:true,
animation:false
}
});

function getPhase(){

if(temp < 0) return "ของแข็ง";

if(temp === 0) return "กำลังหลอมเหลว";

if(temp > 0 && temp < 100) return "ของเหลว";

if(temp === 100) return "กำลังเดือด";

return "แก๊ส";

}

function updateSim(){

time++;

if(temp < 0){
temp += 1 * heatPower;
}
else if(temp === 0 && time < 15){
temp = 0;
}
else if(temp < 100){
temp += 2 * heatPower;
}
else if(temp === 100 && time < 35){
temp = 100;
}
else{
temp += 1;
}

document.getElementById("time").innerText = time;
document.getElementById("temp").innerText = temp;
document.getElementById("power").innerText = heatPower;
document.getElementById("phase").innerText = getPhase();

drawLab();

labels.push(time);
chartData.push(temp);

chart.update();

if(time % 2 === 0){

const row = document.createElement("tr");

row.innerHTML = `
<td>${time}</td>
<td>${temp}</td>
<td>${getPhase()}</td>
`;

document.getElementById("tableBody").appendChild(row);

}

}

function drawLab(){

ctx.clearRect(0,0,500,400);

ctx.strokeRect(170,60,160,230);

if(getPhase().includes("ของแข็ง") || getPhase().includes("หลอม")){

ctx.fillStyle="#9dd9ff";

for(let i=0;i<5;i++){

ctx.fillRect(
190 + (i%2)*50,
220 - i*20,
40,
40
);

}

}

if(getPhase().includes("ของเหลว") || getPhase().includes("เดือด")){

ctx.fillStyle="#4aa3ff";

ctx.fillRect(172,190,156,96);

}

if(getPhase().includes("เดือด") || getPhase().includes("แก๊ส")){

ctx.fillStyle="rgba(180,180,180,0.5)";

for(let i=0;i<12;i++){

ctx.beginPath();

ctx.arc(
200+Math.random()*100,
120-Math.random()*50,
15,
0,
Math.PI*2
);

ctx.fill();

}

}

ctx.fillStyle="orange";

ctx.beginPath();

ctx.moveTo(250,320);
ctx.lineTo(220,380);
ctx.lineTo(280,380);

ctx.closePath();

ctx.fill();

}

function startSim(){

if(!interval){

interval = setInterval(updateSim,1000);

}

}

function pauseSim(){

clearInterval(interval);

interval = null;

}

function resetSim(){

pauseSim();

time = 0;
temp = -10;
heatPower = 1;

labels.length = 0;
chartData.length = 0;

chart.update();

document.getElementById("tableBody").innerHTML="";

document.getElementById("time").innerText=time;
document.getElementById("temp").innerText=temp;
document.getElementById("phase").innerText="ของแข็ง";
document.getElementById("power").innerText=heatPower;

drawLab();

}

function increaseHeat(){

heatPower++;

}

function decreaseHeat(){

if(heatPower>1){

heatPower--;

}

}

drawLab();

</script>
